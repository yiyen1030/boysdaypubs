🪄 NanoBox 示範

將 nanobox.js 和 nanobox.css 放在同一個目錄，然後按照下面的方式使用。

屬性綁定開啟
引入方式
<link rel="stylesheet" href="nanobox.css">
<script src="nanobox.js"></script>
HTML 屬性
<!-- 最簡單 -->
<a href="https://example.com" data-nanobox>開啟</a>

<!-- 帶標題、自訂尺寸 -->
<a href="https://example.com"
   data-nanobox
   data-nanobox-title="標題"
   data-nanobox-width="70vw"
   data-nanobox-height="70vh">開啟</a>
JavaScript API
// 開啟（最簡）
NanoBox.open('https://example.com');

// 帶完整選項
NanoBox.open('https://example.com', {
  title:          '視窗標題',
  width:          '50vw',
  maxWidth:       '760px',
  height:         '80vh',
  closeOnOverlay: true,
  closeOnEsc:     true,
});

// 關閉
NanoBox.close();

// 修改全域預設值
NanoBox.config({ maxWidth: '900px' });

// 動態插入元素後重新綁定
NanoBox.bindAll();
